# --- START OF FILE mcmc_settings.txt ---
# Configuration for Metropolis-Hastings Sampler with Simulated Annealing
#
# This sampler uses:
# 1. Random Step All: Correlated perturbation of all parameters (Adaptive Metropolis)
# 2. Random Step One: Axis-aligned perturbation of a single parameter
# 3. Simulated Annealing: Accepts worse solutions with probability exp(deltaL / T)
# 4. Adaptive Covariance: Learns parameter correlations during sampling

# --- Core MCMC Settings ---

# mcmc_iterations: Total number of MCMC iterations.
# Default: 5000
mcmc_iterations 10000

# thinning: Save 1 sample for every N iterations to reduce autocorrelation.
# Default: 1 (save all)
thinning 50

# report_interval: Frequency (in iterations) for logging progress and checkpointing.
# Default: 100
report_interval 200

# --- Simulated Annealing Parameters ---

# sa_initial_temp: Starting temperature for annealing.
# Higher values allow more exploration of worse solutions initially.
# Default: 100.0
sa_initial_temp 1

# sa_cooling_rate: Geometric cooling factor applied each iteration (T *= cooling_rate).
# Values close to 1.0 = slow cooling, values close to 0.9 = fast cooling.
# Default: 0.999
sa_cooling_rate 1.0

# step_all_prob: Probability of using 'Step All' (global correlated move) vs 'Step One' (local move).
# Higher values favor global exploration, lower values favor local refinement.
# Default: 0.4 (40% global, 60% local)
step_all_prob 0.4

# --- Parallel Candidate Cloud ---

# cloud_size_multiplier: Cloud size = num_threads * multiplier.
# Higher values generate more candidates per iteration for aggressive exploration.
# Default: 2
cloud_size_multiplier 5
